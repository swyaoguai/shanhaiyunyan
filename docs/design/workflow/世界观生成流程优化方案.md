# 世界观生成流程优化方案

## 问题描述

### 问题1：工作流状态面板无法关闭
当执行 `/worldbuild` 命令生成世界观后，右侧会显示"真实执行状态"面板，用户无法手动关闭这个提示，只能等待工作流完成后自动消失。

**当前行为**：
- 工作流面板在 `status !== 'idle'` 时会一直显示
- 没有提供关闭按钮
- 用户体验不佳

### 问题2：世界观生成后显示下载按钮而非直接保存
生成世界观后，系统显示下载按钮让用户下载文件，而不是直接将世界观保存到左侧资料库的"世界设定"分类中。

**当前行为**：
- 世界观生成后作为文件存储在项目目录
- 用户需要点击下载按钮获取文件
- 没有自动同步到前端资料库

**期望行为**：
- 世界观生成后自动保存到资料库
- 用户可以在左侧"世界设定"中直接查看和编辑
- 无需下载文件

## 解决方案

### 方案2：世界观自动保存到资料库

#### 2.1 后端修改：添加世界观保存接口

**修改文件：`novel_agent/web/routes/novel.py`**

```python
@router.post("/world")
async def generate_world(request: GenerateWorldRequest):
    """生成世界观并自动保存到资料库"""
    coordinator = get_coordinator()
    if not coordinator:
        raise HTTPException(status_code=500, detail="Coordinator not initialized")
    
    result = await coordinator.generate_world(
        novel_type=request.novel_type,
        theme=request.theme,
        requirements=request.requirements
    )
    
    # 自动保存到项目数据
    if result and not result.get('error'):
        try:
            # 将世界观数据转换为资料库格式
            world_data = result.get('world', {})
            
            # 构建资料库条目
            world_items = []
            
            # 添加世界名称
            if world_data.get('world_name'):
                world_items.append({
                    'id': f'world_{int(time.time() * 1000)}',
                    'name': world_data['world_name'],
                    'description': '世界名称',
                    'details': world_data.get('world_type', ''),
                    'created_at': datetime.now().isoformat()
                })
            
            # 添加力量体系
            if world_data.get('power_system'):
                world_items.append({
                    'id': f'power_{int(time.time() * 1000) + 1}',
                    'name': '力量体系',
                    'description': '世界的力量体系设定',
                    'details': json.dumps(world_data['power_system'], ensure_ascii=False, indent=2),
                    'created_at': datetime.now().isoformat()
                })
            
            # 添加地理环境
            if world_data.get('geography'):
                world_items.append({
                    'id': f'geo_{int(time.time() * 1000) + 2}',
                    'name': '地理环境',
                    'description': '世界的地理环境设定',
                    'details': json.dumps(world_data['geography'], ensure_ascii=False, indent=2),
                    'created_at': datetime.now().isoformat()
                })
            
            # 添加历史背景
            if world_data.get('history'):
                world_items.append({
                    'id': f'history_{int(time.time() * 1000) + 3}',
                    'name': '历史背景',
                    'description': '世界的历史背景设定',
                    'details': json.dumps(world_data['history'], ensure_ascii=False, indent=2),
                    'created_at': datetime.now().isoformat()
                })
            
            # 添加文化习俗
            if world_data.get('culture'):
                world_items.append({
                    'id': f'culture_{int(time.time() * 1000) + 4}',
                    'name': '文化习俗',
                    'description': '世界的文化习俗设定',
                    'details': json.dumps(world_data['culture'], ensure_ascii=False, indent=2),
                    'created_at': datetime.now().isoformat()
                })
            
            # 保存到worldbuilding数据
            result['world_items'] = world_items
            result['auto_saved'] = True
            
        except Exception as e:
            logger.error(f"Failed to convert world data to library format: {e}")
            result['auto_saved'] = False
    
    return JSONResponse(result)
```

**修改文件：`novel_agent/web/static/app-copilot.js` 或创建新的处理函数**

```javascript
// 新增函数：自动保存世界观到资料库
async function autoSaveWorldToLibrary(workflow) {
    try {
        // 从工作流中获取世界观文件
        const worldFile = workflow.created_files?.find(f => 
            f.kind === 'worldbuilding' || f.path.includes('worldbuilding')
        );
        
        if (!worldFile) return;
        
        // 读取世界观文件内容
        const sessionId = getCurrentCopilotSessionId();
        const res = await apiCall(
            `/api/chat/workflow-file-preview?session_id=${encodeURIComponent(sessionId)}&path=${encodeURIComponent(worldFile.path)}`,
            'GET'
        );
        
        if (!res || !res.content) return;
        
        // 解析世界观数据
        let worldData;
        try {
            worldData = JSON.parse(res.content);
        } catch (e) {
            console.error('Failed to parse world data:', e);
            return;
        }
        
        // 转换为资料库格式并保存
        const worldItems = convertWorldDataToLibraryItems(worldData);
        
        // 保存到本地store
        if (!store.projectData.worldbuilding) {
            store.projectData.worldbuilding = [];
        }
        store.projectData.worldbuilding.push(...worldItems);
        
        // 保存到服务器
        await apiCall('/api/project-data/worldbuilding', 'POST', {
            data: store.projectData.worldbuilding
        });
        
        // 更新@引用数据
        if (typeof updateMentionData === 'function') {
            updateMentionData();
        }
        
        // 显示成功提示
        showToast(`世界观已自动保存到资料库（${worldItems.length}条）`, 'success');
        
        // 如果当前在资料库模块，刷新显示
        if (store.currentModule === 'world') {
            const worldCategory = store.knowledgeCategories.find(c => c.key === 'worldbuilding');
            if (worldCategory && typeof loadDatabase === 'function') {
                loadDatabase(worldCategory.id);
            }
        }
        
    } catch (e) {
        console.error('Failed to auto-save world to library:', e);
        showToast('世界观自动保存失败，请手动保存', 'warning');
    }
}

// 转换世界观数据为资料库条目
function convertWorldDataToLibraryItems(worldData) {
    const items = [];
    const timestamp = Date.now();
    
    // 世界名称
    if (worldData.world_name) {
        items.push({
            id: `world_${timestamp}`,
            name: worldData.world_name,
            description: `世界类型：${worldData.world_type || '未指定'}`,
            details: JSON.stringify(worldData, null, 2),
            created_at: new Date().toISOString()
        });
    }
    
    // 力量体系
    if (worldData.power_system) {
        items.push({
            id: `power_${timestamp + 1}`,
            name: '力量体系',
            description: '世界的力量体系设定',
            details: typeof worldData.power_system === 'string' 
                ? worldData.power_system 
                : JSON.stringify(worldData.power_system, null, 2),
            created_at: new Date().toISOString()
        });
    }
    
    // 地理环境
    if (worldData.geography) {
        items.push({
            id: `geo_${timestamp + 2}`,
            name: '地理环境',
            description: '世界的地理环境设定',
            details: typeof worldData.geography === 'string'
                ? worldData.geography
                : JSON.stringify(worldData.geography, null, 2),
            created_at: new Date().toISOString()
        });
    }
    
    // 历史背景
    if (worldData.history) {
        items.push({
            id: `history_${timestamp + 3}`,
            name: '历史背景',
            description: '世界的历史背景设定',
            details: typeof worldData.history === 'string'
                ? worldData.history
                : JSON.stringify(worldData.history, null, 2),
            created_at: new Date().toISOString()
        });
    }
    
    // 文化习俗
    if (worldData.culture) {
        items.push({
            id: `culture_${timestamp + 4}`,
            name: '文化习俗',
            description: '世界的文化习俗设定',
            details: typeof worldData.culture === 'string'
                ? worldData.culture
                : JSON.stringify(worldData.culture, null, 2),
            created_at: new Date().toISOString()
        });
    }
    
    // 势力阵营
    if (worldData.factions && Array.isArray(worldData.factions)) {
        worldData.factions.forEach((faction, index) => {
            items.push({
                id: `faction_${timestamp + 5 + index}`,
                name: faction.name || `势力${index + 1}`,
                description: faction.description || '势力阵营',
                details: typeof faction === 'string'
                    ? faction
                    : JSON.stringify(faction, null, 2),
                created_at: new Date().toISOString()
            });
        });
    }
    
    return items;
}

// 全局暴露函数
window.autoSaveWorldToLibrary = autoSaveWorldToLibrary;
window.convertWorldDataToLibraryItems = convertWorldDataToLibraryItems;
```

### 方案3：移除下载按钮，改为"查看详情"按钮

如果不想自动保存，可以将下载按钮改为"查看详情"按钮，点击后跳转到资料库：

```javascript
// 在 renderWorkflowFileList 函数中修改按钮
<button type="button" class="copilot-workflow-file-action" data-workflow-view="${path}" title="查看详情">
    <i class="ri-eye-line"></i>
</button>

// 在 bindCopilotWorkflowPanel 中添加处理
const viewBtn = event.target.closest('[data-workflow-view]');
if (viewBtn) {
    const filePath = String(viewBtn.dataset.workflowView || '').trim();
    if (filePath && filePath.includes('worldbuilding')) {
        // 切换到资料库模块
        switchModule('world');
        // 打开世界设定分类
        const worldCategory = store.knowledgeCategories.find(c => c.key === 'worldbuilding');
        if (worldCategory && typeof loadDatabase === 'function') {
            loadDatabase(worldCategory.id);
        }
        showToast('请在左侧资料库中查看世界观设定');
    }
    return;
}
```
